import nltk
from nltk.stem import PorterStemmer
from nltk.corpus import stopwords
import re

class NLPProcessor:
    def __init__(self):
        self.stemmer = PorterStemmer()
        self.stop_words = set(stopwords.words('english')) - {'error', 'fail'}
        self.synonyms = {
            'memory': {'mem', 'ram', 'heap', 'allocation'},
            'crash': {'segfault', 'sigsegv', 'abort'},
            # ... æ›´å¤šåŒä¹‰è¯
        }
    
    def extract_keywords(self, text, max_keywords=15):
        """æ™ºèƒ½å…³é”®è¯æå–"""
        # 1. æå–æŠ€æœ¯æœ¯è¯­
        tech_terms = self._extract_tech_terms(text)
        
        # 2. è¯å¹²æå– + åœç”¨è¯è¿‡æ»¤
        words = text.lower().split()
        processed = [
            self.stemmer.stem(w) 
            for w in words 
            if w not in self.stop_words and len(w) >= 2
        ]
        
        # 3. ç»„åˆå¹¶å»é‡
        return tech_terms + list(set(processed))[:max_keywords]
    
    def expand_with_synonyms(self, keywords):
        """æ‰©å±•åŒä¹‰è¯"""
        expanded = set(keywords)
        for kw in keywords:
            for main, syns in self.synonyms.items():
                if kw == main or kw in syns:
                    expanded.update([main] + list(syns))
        return list(expanded)
    
    def _extract_tech_terms(self, text):
        """æå–æŠ€æœ¯æœ¯è¯­"""
        patterns = {
            'abbreviation': r'\b[A-Z]{2,}\b',
            'hex': r'\b0x[0-9a-f]+\b',
            'function': r'\b\w+\(\)',
        }
        terms = []
        for pattern in patterns.values():
            terms.extend(re.findall(pattern, text))
        return list(set(terms))

# regression_jira_mcp/pattern_learner.py

import re
from collections import Counter

class PatternLearner:
    def __init__(self):
        self.unmatched_errors = []
    
    def record_unmatched(self, log_content, test_name):
        """è®°å½•æœªåŒ¹é…çš„é”™è¯¯"""
        error_lines = [
            line for line in log_content.split('\n')
            if any(w in line.lower() for w in ['error', 'fail', 'fatal'])
        ]
        self.unmatched_errors.extend(error_lines)
    
    def analyze_patterns(self, min_frequency=3):
        """åˆ†æå¹¶å»ºè®®æ–°æ¨¡å¼"""
        # æå–3-gramçŸ­è¯­
        phrases = Counter()
        for line in self.unmatched_errors:
            words = line.split()
            for i in range(len(words) - 2):
                phrase = ' '.join(words[i:i+3])
                phrases[phrase] += 1
        
        # ç”Ÿæˆå»ºè®®
        suggestions = []
        for phrase, freq in phrases.items():
            if freq >= min_frequency:
                regex = self._generalize(phrase)
                suggestions.append({
                    'pattern': regex,
                    'frequency': freq,
                    'confidence': 'high' if freq >= 10 else 'medium'
                })
        
        return suggestions
    
    def _generalize(self, text):
        """æ³›åŒ–ä¸ºæ­£åˆ™è¡¨è¾¾å¼"""
        regex = text
        regex = re.sub(r'\b\d+\b', r'\\d+', regex)
        regex = re.sub(r'\b0x[0-9a-f]+\b', r'0x[0-9a-f]+', regex)
        return regex
